def system(cmd)
  stdout.clear
  stderr.clear
  status = Open4::spawn(cmd, 'stdout' => stdout, 'stderr' => stderr, 'stdin' => stdin, 'ignore_exit_failure' => true)
  @exitstatus = status.exitstatus
end
