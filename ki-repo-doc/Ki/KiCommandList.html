def execute(ctx, args)
  commands = KiCommand::CommandRegistry.find(KiCommand::CommandPrefix[0..-2])
  commands.each do |id, service_class|
    puts "  #{id[KiCommand::CommandPrefix.size..-1]}: #{service_class.new.summary}"
  end
end
